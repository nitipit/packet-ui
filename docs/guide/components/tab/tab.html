SCSS

@use 'path/to/_ui.scss';

pkt-tab {
    @include ui.tab(
        $button-color: grey,
        $highlight-color: orange)
}
Argument Description Example Value
$button-color Button color red blue
$highlight-color Bottom border color when active red blue
