@use 'path/to/_ui.scss';

    @include ui.slider(
        $thumb-color: color.adjust(white, $lightness: -10%),
        $thumb-color-active: orange,
        $thumb-width: 1.2em,
        $thumb-height: 1.2em,
        $thumb-border: 1px solid,
        $thumb-border-radius: 50%,
        $bar-color: color.adjust(white, $lightness: -20%),
        $bar-height: 0.25em
    )
}
